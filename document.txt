type Department @model {
  id: ID!
  name: String!
  description: String
  logoUri: String
  createdAt: AWSDateTime
  jobs: [Job] @hasMany(indexName: "DepartmentJobs", fields: ["id"])
}

type Job @model {
  id: ID!
  title: String!
  description: String!
  duties: String!
  requirements: String!
  niceToHave: String!
  benefits: String!
  minSalary: Int
  maxSalary: Int
  jobType: JobType!
  jobMode: JobMode!
  location: String
  createdAt: AWSDateTime
  jobDepartmentId: ID!
  department: Department! @belongsTo(fields: ["departmentId"])
  jobDepartmentId: ID! @index(name: "DepartmentJobs", sortKeyFields: ["id"])
   applicants: [Applicant]
   @hasMany(indexName: "JobApplicants", fields: ["jobId"])
}

 jobDepartmentId and jobDepartmentId is the same
